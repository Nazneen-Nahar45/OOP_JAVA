import java.util.ArrayList;
import java.util.Scanner;

/**
 * UapBazar
 */
public class UapBazar {


    /**
     * @param args
     */
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        String name;
        String id;
        String category;
        double price;

        ArrayList<Product> e = new ArrayList<>();
     
        int choice;
        String am;
        double amount;

        do {
            System.out.println("1.Add Product :");
            System.out.println("2.update price :");
            System.out.println("3.Discount Price :");
            System.out.println("4.Product List :");

            System.out.println("Press any Option : ");
            choice = input.nextInt();
            switch (choice) {
                case 1:
                    System.out.println("Enter Product Name : ");
                    name = input.next();
                    System.out.println("Enter prodect id: ");
                    id = input.next();
                    System.out.println("Enter Product Price :");
                    price = input.nextDouble();
                    System.out.println("Enter catagory: ");
                    category = input.next();

                    e.add(new Product(name,price,id,category));
                  
                    break;
                case 2:
                    System.out.println("Enter Id : ");
                    am = input.next();
                    System.out.println("Enter new price : ");
                    amount = input.nextDouble();
                    for (int i = 0; i < e.size(); i++) {
                        if (e.get(i).id.equals(am)) {
                            e.get(i).UpdatePrice(amount);
                            break;
                        }
                    }

                    break;
                    case 3:
                    System.out.println("Enter Id : ");
                    am = input.next();
                    System.out.println("Enter discount price : ");
                    amount = input.nextDouble();
                    for (int i = 0; i < e.size(); i++) {
                        if (e.get(i).id.equals(am)) {
                           e.get(i).GetDiscountedPrice(amount);
                            break;
                        }
                    }

                    break;
                    case 4 :
                    for(int i=0; i<e.size(); i++)
                    {
                      
                      
                            e.get(i).display();
                        
                    }
                    break;
                default:
                    break;
            }

        } while (choice!=0);
        input.close();
    }

  
    }
