public class items {

    public String name;
    public String id;
    public String category;
    public double price;
    public int quantity;

    public items(String name, String id, String category, double price, int quantity) {
        this.name = name;
        this.id = id;
        this.category = category;
        this.price = price;
        this.quantity = quantity;

    }

    public void updatePrice(double newPrice) {
        price = newPrice;
    }

    public void display() {

        System.out.println("Name"+"\n"+ name + "id"+"\n" + id + "Category"+"\n" + category + "Price"+"\n" + price + "Quantity"+"\n"+ quantity);
    }

}
